Ressources files#

This section contains the template files to restore a broken package. The structure of the complete package is described in the Project Structure section.

Il the various files please replace the following placeholders:

  • {author_name}: The name(s) of the author of the package. (ex: Artezaru)

  • {author_email}: The email of the author of the package. (ex: artezaru.github@proton.me)

  • {github_repo}: The GitHub repository of the package. (ex: Artezaru/pyprogen.git)

  • {gitpage_doc}: The GitHub page of the package. (ex: https://artezaru.github.io/pyprogen/)

  • {package_name}: The name of the package. (ex: pyprogen)

  • {*_dir}: The path of the corresponding directory.

  • {*_file}: The path of the corresponding file.

Also replace the {{ and }} by { and } respectively (in setup.py and conf.py files).

setup.py#

import os
from setuptools import setup, find_packages

# 0. Read the contents of your README.md file to use as the long description
def read_long_description():
    with open("{project_readme_file}", "r") as file:
        long_description = file.read()
    return long_description

# 1. Read the contents of your requirements.txt file.
def read_requirements():
    with open("{package_requirements_file}", "r") as file:
        # Remove comments and empty lines
        return [line.strip() for line in file.readlines() if line.strip() and not line.startswith('#')]

# 2. Read the version in the __version__.py file
def read_version():
    version_file = "{package_version_file}"
    with open(version_file, "r") as file:
        exec(file.read()) 
    return locals()["__version__"]

# 3. Setup the package
setup(
    name="{package_name}",
    version=read_version(), # Update the version as necessary
    author="{author_name}",
    author_email="{author_email}",
    description="",  # Short description of the package
    long_description=read_long_description(), # Update the long description as necessary
    long_description_content_type="text/markdown",  # Format of the long description
    url="{github_repo}",  # URL to the package's homepage
    packages=find_packages(exclude=["tests", "tests.*"]),  # Automatically find packages in the directory
    classifiers=[
        "Programming Language :: Python :: 3",
        "License :: OSI Approved :: Apache Software License",
        "Operating System :: OS Independent",
    ],
    python_requires='>=3.6',  # Minimum Python version required
    install_requires=read_requirements(),
    include_package_data=True,  # Include package data as specified in MANIFEST.in
    package_data={{
        '{package_name}': ['ressources/*'],  # Include all files in ressources/
    }},
    entry_points={{
        'console_scripts': [
            '{package_name} = {package_name}.__main__:main',  # Define the {package_name} command to run __main__.main()
        ],
    }},
)

# 0. Setting directories for the documentation
SPHINXOPTS    ?=
SPHINXBUILD   ?= sphinx-build
SOURCEDIR     = {documentation_source_dir}
BUILDDIR      = {documentation_build_dir}

# 1. Default help command to list available Sphinx options
help:
	@$(SPHINXBUILD) -M help "$(SOURCEDIR)" "$(BUILDDIR)" $(SPHINXOPTS) $(O)

.PHONY: help Makefile

# 2. Generate HTML documentation
html:
	$(SPHINXBUILD) -b html $(SOURCEDIR) $(BUILDDIR)/html

# 3. Generate LaTeX PDF documentation
latexpdf:
	$(SPHINXBUILD) -b latex $(SOURCEDIR) $(BUILDDIR)/latex
	cd $(BUILDDIR)/latex && pdflatex {package_name}.tex && pdflatex {package_name}.tex

# 4. Clean the documentation
clean:
	@$(SPHINXBUILD) -M clean "$(SOURCEDIR)" "$(BUILDDIR)" $(SPHINXOPTS) $(O);
	cd $(BUILDDIR); git worktree add -f html gh-pages; cd html; rm -rf *
